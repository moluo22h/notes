```bash
server{
	listen	80;
	listen 9999;
	server_name www.moluo.hello www.moluo.hi;
	root /data/www;
	index index.html index.htm;
	access_log /var/log/nginx/access_imooc.log imooc;
	location /{
		rewrite ^(.*).html$ /index.html;
	}
```
